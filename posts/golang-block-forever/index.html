    Golang: Block forever

    Apr 27, 2020
    #golang #trick #note

    Sometimes, you want to block the current goroutine when allowing others to continue. Here is some tricks I’ve collected:

    1. References #

    Firstly give them some credits:

    1. https://blog.sgmansfield.com/2016/06/how-to-block-forever-in-go/
    2. https://pliutau.com/different-ways-to-block-go-runtime-forever/

    NOTE: I run these with Golang 1.12

    2. The original # 

     1package main
 2
 3import "fmt"
 4
 5func show() {
 6    for i := 1; i < 9696969; i++ {
 7        time.Sleep(1000)
 8        fmt.Println(i)
 9    }
10}
11
12func main() {
13    go show() // The main goroutine is exited before the show() be done.
14    fmt.Println("OK we're done")
15}

    3. Bad - An empty infinite loop # 

     1package main
 2
 3import (
 4    "fmt"
 5    "time"
 6)
 7
 8func forever() {
 9    for {
10        // Empty, just do nothing
11    }
12}
13
14func show() {
15    for i := 1; i < 9696969; i++ {
16        time.Sleep(5 * time.Second)
17        fmt.Println(i)
18    }
19}
20
21func main() {
22    go show()
23    forever()
24    fmt.Println("OK we're done")
25}

    An infinite loop here is a busy loop that does nothing except burn CPU time.

    4. Good - Busy blocking # 

     1package main
 2
 3import (
 4    "fmt"
 5    "runtime"
 6    "time"
 7)
 8
 9func forever() {
10    for {
11        runtime.Gosched()
12    }
13}
14
15func show() {
16    for i := 1; i < 9696969; i++ {
17        time.Sleep(5 * time.Second)
18        fmt.Println(i)
19    }
20}
21
22func main() {
23    go show()
24    forever()
25    fmt.Println("OK we're done")
26}

    It will reduce your CPU usage but it isn’t the preferable solution.

    5. Good - Waiting on itself #

    We wait but we never done XD

     1package main
 2
 3import (
 4    "fmt"
 5    "sync"
 6    "time"
 7)
 8
 9func forever() {
10    wg := sync.WaitGroup{}
11    wg.Add(1)
12    wg.Wait()
13}
14
15func show() {
16    for i := 1; i < 9696969; i++ {
17        time.Sleep(5 * time.Second)
18        fmt.Println(i)
19    }
20}
21
22func main() {
23    go show()
24    forever()
25    fmt.Println("OK we're done")
26}

    7. Good - Double locking # 

     1package main
 2
 3import (
 4    "fmt"
 5    "sync"
 6    "time"
 7)
 8
 9func forever() {
10    m := sync.Mutex{} // Same with sync.RWMutex
11    m.Lock()
12    m.Lock()
13}
14
15func show() {
16    for i := 1; i < 9696969; i++ {
17        time.Sleep(5 * time.Second)
18        fmt.Println(i)
19    }
20}
21
22func main() {
23    go show()
24    forever()
25    fmt.Println("OK we're done")
26}

    8. Good - Reading an Empty Channel # 

     1package main
 2
 3import (
 4    "fmt"
 5    "time"
 6)
 7
 8func forever() {
 9    c := make(chan struct{})
10    <-c
11}
12
13func show() {
14    for i := 1; i < 9696969; i++ {
15        time.Sleep(5 * time.Second)
16        fmt.Println(i)
17    }
18}
19
20func main() {

    9. Good - Self produce-and-consume # 

     1package main
 2
 3import (
 4    "fmt"
 5    "time"
 6)
 7
 8func forever() {
 9    c := make(chan struct{}, 1)
10    for {
11        select {
12        case <-c:
13        case c <- struct{}{}:
14        }
15    }
16}
17
18func show() {
19    for i := 1; i < 9696969; i++ {
20        time.Sleep(5 * time.Second)
21        fmt.Println(i)
22    }
23}
24
25func main() {
26    go show()
27    forever()
28    fmt.Println("OK we're done")
29}
